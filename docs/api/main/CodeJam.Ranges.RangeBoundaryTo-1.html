public struct RangeBoundaryTo<T> : IEquatable<RangeBoundaryTo<T>>, IComparable<RangeBoundaryTo<T>>, IComparable<RangeBoundaryFrom<T>>, IComparable<T>, IComparable, IFormattable
    Type Parameters
    Name Description
    T

    The type of the value. Should implement System.IComparable<T> or System.IComparable.
    Remarks

    Heavy tuned to be as fast as it is possible. The order of borders is the following: '∅' < '+∞' < 'a)' < '[a' == 'a]' < '(a' < '-∞'.
